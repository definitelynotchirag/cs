#include <iostream>
using namespace std;

class Queue
{
public:
  string arr[10];
  int front = 0;
  int rear = 0;

  void clear()
  {
    for (int i = 0; i < 10; i++)
    {
      string e = "__";
      arr[i] = e;
    }
  }

  bool isempty()
  {
    if (front == rear)
      return 1;
    else
      return 0;
  }

  void addjob(string name)
  {
    if (rear == 9)
    {
      arr[rear++] = name;
      cout << "Now the Job Queue is Full" << endl;
    }
    else if (rear > 9)
    {
      cout << "Queue Over flow!!!!!!!!!!!!" << endl;
      // front = 10;
    }
    else
    {
      arr[rear++] = name;
    }
  }

  void del()
  {
    if (isempty())
    {
      cout << "Queue is underflown" << endl;
    }
    else
    { 
      cout<<"Job To be Deleted :"<<arr[front]<<endl;
      for(int i = 0 ; i < 10-1 ; i++){
        arr[i]=arr[i+1];
      }
    }
  }

  void display()
  {
    for (int i = 0; i < 10; i++)
    {
      cout << arr[i] << " ";
    }
    cout << endl;
  }
};

int main()
{
  Queue j;
  int ch;
  j.clear();
  cout << "Enter the choice :";
  cin >> ch;
  while (true)
  {
    if (ch == 1)
    {
      string n;
      cout << "Enter the Applicants Name :";
      cin >> n;
      j.addjob(n);
      j.display();
    }
    else if (ch == 2)
    {
      j.del();
      j.display();
    }
    else
    {
      cout << "Final Queue :";
      j.display();
      break;
    }
    cout << "Enter the choice :";
    cin >> ch;
  }
    
}
